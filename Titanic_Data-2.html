#This is the function to fill in the missing age values by using the approximations based off of Pclass
def age_approx(cols): 
    Age = cols[0]
    Pclass = cols[1]
    if pd.isnull(Age): 
        if Pclass == 1:
            return 38 
        elif Pclass == 2: 
            return 30
        else:
            return 23
    else:
        return Age
